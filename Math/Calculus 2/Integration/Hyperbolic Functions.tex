\section{Standard Hyperbolic Functions}

\subsection{Standard Functions}
\begin{flalign*}
    & \int \sinh(ax)dx = \frac{\cosh(ax)}{a}+C\\
    & \int \cosh(ax)dx = \frac{\sinh(ax)}{a}+C\\
    & \int \tanh(ax)dx = \frac{\ln|\cosh(ax)|}{a}+C\\
    & \int \csch(ax)dx = \frac{\ln |\tanh(\frac{ax}{2})|}{a}+C\\
    & \int \sech (ax)dx = \frac{\arctan(\sinh(ax))}{a} + C\\
    & \int \coth(ax)dx = \frac{\ln|\sinh(ax)|}{a}+C\\
\end{flalign*}

\subsection{Inverse functions}
\begin{flalign*}
    & \int \arcsin(ax)dx = x \arcsin(ax) + \frac{\sqrt{1-a^2x^2}}{a} + C\\
\end{flalign*}
